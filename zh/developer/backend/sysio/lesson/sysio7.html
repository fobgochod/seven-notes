cd /proc/$$/fd
# fd8的输入输出都指向socket
exec 8<> /dev/tcp/www.baidu.com/80
# echo输出重定向到fd8  文件描述符需在>后加上&
# HTTP协议规定Request的最小写法 'GET / HTTP/1.0\n'
echo -e 'GET / HTTP/1.0\n' >& 8
# cat输入重新定向到fd8
cat 0<& 8

# 关闭
exec 8<& -

# 网卡信息
